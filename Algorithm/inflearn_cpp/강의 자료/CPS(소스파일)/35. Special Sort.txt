#include<stdio.h>
void Special_Sort(int temp[], int len){
	int i, j, t;
	for(i=0; i<len; i++){
		for(j=0; j<len-i-1; j++){
			if(temp[j]>0 && temp[j+1]<0){
				t=temp[j];
				temp[j]=temp[j+1];
				temp[j+1]=t;
			}
		}	
	}
}

int main(){
	freopen("input.txt", "rt", stdin);
	freopen("output.txt", "wt", stdout);
	int n, i;
	scanf("%d", &n);
	int *a=new int[n];
	for(i=0; i<n; i++){
		scanf("%d", &a[i]);
	}
	Special_Sort(a, n);
	for(i=0; i<n; i++){
		printf("%d ", a[i]);
	}
	delete[] a;
	return 0;
}
